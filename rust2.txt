use std::io;

fn main() {
  
    println!("Введите строку:");

    // Считываем строку из стандартного ввода
    let mut sentence = String::new();
    io::stdin().read_line(&mut sentence)
        .expect("Ошибка при чтении строки");

    // Создаем битовую маску; каждый бит (в младших 26 битах) соответствует букве a-z
    let mut letters: u32 = 0;

    // Проходим по каждому символу строки
    for c in sentence.chars() {
        if c >= 'a' && c <= 'z' {
            // Смещение относительно 'a' определяет позицию бита,
            // который устанавливаем в переменной letters
            letters |= 1 << (c as u32 - 'a' as u32);
        }
    }

    // Если все 26 битов установлены, то letters равен (1 << 26) - 1
    // и строка является панграммой
    if letters == (1 << 26) - 1 {
        println!("true");
    } else {
        println!("false");
    }
}
