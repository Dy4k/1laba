fun main() {
    // Выводим приглашение ко вводу строки
    println("Введите строку:")

    // Считываем строку с консоли (если readLine() вернет null, используем пустую строку)
    val sentence = readLine() ?: ""

    // Используем битовую маску: каждый из 26 битов соответствует букве a-z
    var letters = 0

    // Проходим по каждому символу строки
    for (c in sentence) {
        if (c in 'a'..'z') {
            // Вычисляем позицию буквы (от 'a') и устанавливаем соответствующий бит
            letters = letters or (1 shl (c - 'a'))
        }
    }

    // Если все 26 битов установлены, letters должно быть равно (1 shl 26) - 1
    if (letters == (1 shl 26) - 1)
        println("true")
    else
        println("false")
}
