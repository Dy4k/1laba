puts "Введите строку:"

# Считываем введённую строку
sentence = gets.chomp

# Инициализируем переменную для битовой маски,
# где 26 младших битов будут представлять наличие букв a-z.
letters = 0

# Обходим каждый символ строки
sentence.each_char do |c|
  # Проверяем, что символ является строчной буквой от 'a' до 'z'
  if c >= 'a' && c <= 'z'
    # Вычисляем смещение (относительно ascii-кода 'a')
    # и устанавливаем соответствующий бит в переменной letters.
    letters |= 1 << (c.ord - 'a'.ord)
  end
end

# Если все 26 битов установлены (число равно (1 << 26) - 1),
# значит строка содержит все буквы английского алфавита.
if letters == (1 << 26) - 1
  puts "true"
else
  puts "false"
end
