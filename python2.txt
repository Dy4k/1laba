def main():
    # Выводим приглашение и считываем всю строку, включая пробелы
    sentence = input("Введите строку:\n")
    
    # Используем битовую маску: каждый бит соответствует одной букве (a-z)
    letters = 0
    
    # Проходим по каждому символу строки
    for c in sentence:
        if 'a' <= c <= 'z':
            # Устанавливаем бит, соответствующий букве c
            letters |= (1 << (ord(c) - ord('a')))
    
    # Если установлены все 26 битов, то строка является панграммой
    if letters == (1 << 26) - 1:
        print("true")
    else:
        print("false")

main()
