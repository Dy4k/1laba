package main

import (
    "bufio" // для буферизованного ввода с консоли.
    "fmt"   // для форматированного ввода и вывода.
    "os"  // для доступа к операционной системе (например, стандартный ввод).
    "strings" //для работы со строками (обрезка пробелов и т.д.).

)

func main() {
    // Выводим приглашение ко вводу
    fmt.Println("Введите строку:")

    // Считываем строку с консоли
    reader := bufio.NewReader(os.Stdin)
    sentence, err := reader.ReadString('\n')
    if err != nil {
        fmt.Println("Ошибка чтения:", err)
        return
    }
    // Удаляем лишние пробельные символы
    sentence = strings.TrimSpace(sentence)

    // Инициализируем переменную для битовой маски (26 младших битов для букв a-z)
    var letters int = 0

    // Проходим по каждому символу строки
    for _, c := range sentence {
        if c >= 'a' && c <= 'z' {
            // Устанавливаем бит, соответствующий букве c
            letters |= 1 << (c - 'a')
        }
    }

    // Проверяем, что все 26 битов установлены (число должно быть равно (1<<26)-1)
    if letters == (1<<26)-1 {
        fmt.Println("true")
    } else {
        fmt.Println("false")
    }
}
