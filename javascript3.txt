const readline = require('readline');

// Функция для вычисления суммы цифр числа
function sumDigits(n) {
    n = Math.abs(n); 
    let sum = 0;
    while (n > 0) {
        sum += n % 10;
        n = Math.floor(n / 10); 
    }
    return sum;
}

const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout,
    terminal: false
});

function main() {
    console.log("Введите числа:");

    rl.on('line', (line) => {
        // Разбиваем строку по пробельным символам
        const parts = line.trim().split(/\s+/);
        let Numbers = [];

        for (const part of parts) {
            const num = parseInt(part, 10);
            if (!isNaN(num)) {
                // Рассматриваем только числа с минимум двумя цифрами и с суммой цифр меньше 10
                if (Math.abs(num) >= 10 && sumDigits(num) < 10) {
                    Numbers.push(num);
                }
            }
        }

        if (Numbers.length > 0) {
            Numbers.sort((a, b) => a - b);
            console.log("Числа, сумма цифр которого состоит из одного десятичного разряда:");
            console.log(Numbers.join(" "));
        } else {
            console.log("Нет подходящих чисел");
        }

        rl.close();
    });
}

main();
